# 全局自定义指令  
Vue.directive('focus', {  
&emsp;// 当绑定元素插入到 DOM 中。  
&emsp;inserted: function (el) {  
&emsp;&emsp;// 聚焦元素  
&emsp;&emsp;el.focus()  
&emsp;}  
})  
# 仅在某实例中使用的指令
directives: {  
&emsp;// 注册一个局部的自定义指令 v-focus  
&emsp;focus: {   // 这里focus是指令名
&emsp;&emsp;// 指令的定义  
&emsp;&emsp;inserted: function (el) {  
&emsp;&emsp;&emsp;// 聚焦元素  
&emsp;&emsp;&emsp;el.focus()  
&emsp;&emsp;}  
&emsp;}  
}  
注意，注册时指令名不加v-，使用时：v-指令名='表达式'
